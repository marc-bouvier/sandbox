git clone git@github.com:spring-projects/spring-restdocs.git
git checkout v2.0.5.RELEASE
sdk use java 8.0.272-zulu
sdk use gradle 4.9
cd cd samples/web-test-client
gradle build
# Ouvrir le html généré par le Asciidoc

# Ouvrir le test
